[docs]    def get_mean(self, X, Psi=None, params=None, bias=None):
        """
        calculate mean values

        Parameters
        ==========
        X: numpy.ndarray
            input as an N-by-d matrix
        Psi: numpy.ndarray
            feature maps ``Psi(X)`` as an N-by-l matrix
            (default: self.get_basis(X))
        params: numpy.ndarray
            weight as a vector with size l
            (default: self.params)
        bias: float
            (default: self.bias)

        Returns
        =======
        numpy.ndarray
            Psi * params + bias

        """
        if params is None:
            params = np.copy(self.params)

        if bias is None:
            bias = np.copy(self.bias)

        if Psi is None:
            Psi = self.get_basis(X)

        return Psi.dot(params) + bias
